72 inline uint32_t __attribute__ ((always_inline)) neorv32_cpu_store_conditional(uint32_t addr, uint32_t wdata) {
73 
74 #if defined __riscv_atomic || defined __riscv_a
75  register uint32_t reg_addr = addr;
76  register uint32_t reg_data = wdata;
77  register uint32_t reg_status;
78 
79  asm volatile ("sc.w %[status], %[da], (%[ad])" : [status] "=r" (reg_status) : [da] "r" (reg_data), [ad] "r" (reg_addr));
80 
81  return reg_status;
82 #else
83  return 1; // always failing
84 #endif
85 }
